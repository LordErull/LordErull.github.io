<script>
let step=0;

let texts=[
  "Aku ada sesuatu nih 😏",
  "Serius ini penting...",
  "Siap ya...",
  "Selamat ulang tahun 🎉"
];

let texts2=[
  "Semoga panjang umur",
  "Semoga bahagia selalu",
  "Semoga semua tercapai"
];

function mulai(){
  let nama=document.getElementById("nama").value;
  if(!nama) return alert("Isi nama dulu");

  start.classList.add("hidden");
  main.classList.remove("hidden");

  namaTampil.innerHTML=nama;

  ketik();
}

function ketik(){
  let t=texts[step];
  let i=0;
  text.innerHTML="";

  let int=setInterval(()=>{
    text.innerHTML+=t[i];
    i++;
    if(i>=t.length){
      clearInterval(int);
      if(step<texts.length-1){
        setTimeout(()=>{step++;ketik();},1000);
      }else showBtn();
    }
  },40);
}

function showBtn(){
  btnArea.innerHTML="";
  let b=document.createElement("button");
  b.innerText="Lanjut ➜";
  b.onclick=()=>{main.classList.add("hidden");tanya.classList.remove("hidden");};
  btnArea.appendChild(b);
}

function hindar(btn){
  btn.style.left=Math.random()*100+"px";
  btn.style.top=Math.random()*50+"px";
  jahil.innerHTML="Eits ga bisa 😝";
}

function jawabYa(){
  tanya.classList.add("hidden");
  main.classList.remove("hidden");
  step=0;
  lanjut2();
}

function lanjut2(){
  let t=texts2[step];
  let i=0;
  text.innerHTML="";

  let int=setInterval(()=>{
    text.innerHTML+=t[i];
    i++;
    if(i>=t.length){
      clearInterval(int);
      if(step<texts2.length-1){
        setTimeout(()=>{step++;lanjut2();},1000);
      }else{
        setTimeout(()=>{
          main.classList.add("hidden");
          akhir.classList.remove("hidden");
        },1000);
      }
    }
  },40);
}

function kirimWA(){
  let p=pesan.value;
  if(!p) return alert("Isi pesan dulu");

  window.open("https://wa.me/6281234567890?text="+encodeURIComponent(p));
}
